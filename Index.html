Temperature Conversion

This Ruby program will prompt the user for a temperature in degrees Celsius and let the user know what the corresponding temperature is in Fahrenheit. 


def temp_conversion(val1)  
    val2 = (val1.to_f * 9 / 5) + 32     
    #(0°C × 9/5) + 32 = 32°F - formula for temp conversion  
    response_cm = "The temperature is #{val2} degrees Fahrenheit"  
    return response_cm 
end

puts "Please enter degrees in Celsius: "   
val1 = gets.chomp 

puts temp_conversion(val1)


def instance(val1)  
    right_most_val = val1 % 10      
    if val1 == 11 || val1 == 12 || val1 == 13        
        return "#{val1}th" 
    end
    if right_most_val == 1    
        return "#{val1}st" 
    elsif right_most_val == 2 
        return "#{val1}nd" 
    elsif right_most_val == 3             
        return "#{val1}rd"          
    else    
        return "#{val1}th"     
    end     
end

puts "Enter a number"       
val1 = gets.chomp.to_i           

puts instance(val1)

Foobar

This Ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

    
puts "How many items do you want to see?"   
val1 = gets.chomp.to_i    
    
list1 = []    
(1..val1).each do |number|    
    if number % 3 == 0 && number % 5 == 0     
       list1 << "foobar" 
    elsif number % 3 == 0       
       list1 << "foo"   
    elsif number % 5 == 0       
       list1 << "bar"   
    else
       list1 << number        
    end  
end

puts "#{list1}"
